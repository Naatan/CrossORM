  41.         $this->_sections[0] = array('title' => 'Overview', 'url' => 'overview-summary.html');

  50.

  73.                         }

  84.                 }
  85.             }
  86.         }
  87.         $deprecatedFunctions = array();
  88.         $functions =& $rootDoc->functions();

  91.                 if ($function->tags('@deprecated')) {

  93.                 }
  94.             }
  95.         }
  96.

  109.                 echo '<li><a href="#deprecated_class">Deprecated Classes</a></li>';
  110.             }
  111.             if ($deprecatedFields) {
  112.                 echo '<li><a href="#deprecated_field">Deprecated Fields</a></li>';
  113.             }
  114.             if ($deprecatedMethods) {
  115.                 echo '<li><a href="#deprecated_method">Deprecated Methods</a></li>';
  116.             }
  117.             if ($deprecatedGlobals) {
  118.                 echo '<li><a href="#deprecated_global">Deprecated Globals</a></li>';
  119.             }
  120.             if ($deprecatedFunctions) {
  121.                 echo '<li><a href="#deprecated_function">Deprecated Functions</a></li>';
  122.             }
  123.             echo "</ul>\n";
  124.         }
  125.  
  126.         if ($deprecatedClasses) {
  127.             echo '<table id="deprecated_class" class="detail">', "\n";
  128.             echo '<tr><th colspan="2" class="title">Deprecated Classes</th></tr>', "\n";
  129.             foreach($deprecatedClasses as $class) {
  130.                 $textTag =& $class->tags('@text');
  131.                 echo '<tr><td class="name"><a href="', $class->asPath(), '">', $class->qualifiedName(), '</a></td>';
  132.                 echo '<td class="description">';
  133.                 if ($textTag) echo strip_tags($this->_processInlineTags($textTag, TRUE), '<a><b><strong><u><em>');
  134.                 echo "</td></tr>\n";
  135.             }
  136.             echo "</table>\n\n";
  137.         }
  138.  
  139.         if ($deprecatedFields) {
  140.             echo '<table id="deprecated_field" class="detail">', "\n";
  141.             echo '<tr><th colspan="2" class="title">Deprecated Fields</th></tr>', "\n";
  142.             foreach ($deprecatedFields as $field) {
  143.                 $textTag =& $field->tags('@text');
  144.                 echo "<tr>\n";
  145.                 echo '<td class="name"><a href="', $field->asPath(), '">', $field->qualifiedName(), "</a></td>\n";
  146.                 echo '<td class="description">';
  147.                 if ($textTag) echo strip_tags($this->_processInlineTags($textTag, TRUE), '<a><b><strong><u><em>');
  148.                 echo "</td>\n";
  149.                 echo "</tr>\n";
  150.             }
  151.             echo "</table>\n\n";
  152.         }
  153.  
  154.         if ($deprecatedMethods) {
  155.             echo '<table id="deprecated_method" class="detail">', "\n";
  156.             echo '<tr><th colspan="2" class="title">Deprecated Methods</th></tr>', "\n";
  157.             foreach($deprecatedMethods as $method) {
  158.                 $textTag =& $method->tags('@text');
  159.                 echo "<tr>\n";
  160.                 echo '<td class="name"><a href="', $method->asPath(), '">', $method->qualifiedName(), "</a></td>\n";
  161.                 echo '<td class="description">';
  162.                 if ($textTag) echo strip_tags($this->_processInlineTags($textTag, TRUE), '<a><b><strong><u><em>');
  163.                 echo "</td>\n";
  164.                 echo "</tr>\n";
  165.             }
  166.             echo "</table>\n\n";
  167.         }
  168.  
  169.         if ($deprecatedGlobals) {
  170.             echo '<table id="deprecated_global" class="detail">', "\n";
  171.             echo '<tr><th colspan="2" class="title">Deprecated Globals</th></tr>', "\n";
  172.             foreach($deprecatedGlobals as $global) {
  173.                 $textTag =& $global->tags('@text');
  174.                 echo "<tr>\n";
  175.                 echo '<td class="name"><a href="', $global->asPath(), '">', $global->qualifiedName(), "</a></td>\n";
  176.                 echo '<td class="description">';
  177.                 if ($textTag) echo strip_tags($this->_processInlineTags($textTag, TRUE), '<a><b><strong><u><em>');
  178.                 echo "</td>\n";
  179.                 echo "</tr>\n";
  180.             }
  181.             echo "</table>\n\n";
  182. 		}
  183.  
  184.         if ($deprecatedFunctions) {
  185.             echo '<table id="deprecated_function" class="detail">', "\n";
  186.             echo '<tr><th colspan="2" class="title">Deprecated Functions</th></tr>', "\n";
  187.             foreach($deprecatedFunctions as $function) {
  188.                 $textTag =& $function->tags('@text');
  189.                 echo "<tr>\n";
  190.                 echo '<td class="name"><a href="', $function->asPath(), '">', $function->qualifiedName(), "</a></td>\n";
  191.                 echo '<td class="description">';
