  28. 	/** Build the element index.
  29. 	 *
  30. 	 * @param Doclet doclet
  31. 	 */
  32. 	function indexWriter(&$doclet)
  33.     {
  34.  
  35. 		parent::HTMLWriter($doclet);
  36.  
  37. 		//$this->_id = 'definition';
  38.  
  39. 		$rootDoc =& $this->_doclet->rootDoc();
  40.  
  41.         $this->_sections[0] = array('title' => 'Overview', 'url' => 'overview-summary.html');
  42.         $this->_sections[1] = array('title' => 'Namespace');
  43.         $this->_sections[2] = array('title' => 'Class');
  44.         //$this->_sections[3] = array('title' => 'Use');
  45.         $this->_sections[4] = array('title' => 'Tree', 'url' => 'overview-tree.html');
  46.         if ($doclet->includeSource()) $this->_sections[5] = array('title' => 'Files', 'url' => 'overview-files.html');
  47.         $this->_sections[6] = array('title' => 'Deprecated', 'url' => 'deprecated-list.html');
  48.         $this->_sections[7] = array('title' => 'Todo', 'url' => 'todo-list.html');
  49.         $this->_sections[8] = array('title' => 'Index', 'selected' => TRUE);
  50.  
  51.         $classes =& $rootDoc->classes();
  52.         if($classes == NULL) $classes = array();
  53.  
  54.         $methods = array();
  55.         foreach ($classes as $class) {
  56.             foreach ($class->methods(TRUE) as $name => $method) {
  57.                 $methods[$class->name().'::'.$name] = $method;
  58.             }
  59.         }
  60.         if($methods == NULL) $methods = array();
  61.  
  62.         $functions =& $rootDoc->functions();
  63.         if($functions == NULL) $functions = array();
  64.  
  65.         $globals =& $rootDoc->globals();
  66.         if($globals == NULL) $globals = array();
  67.  
  68.         $elements = array_merge($classes, $methods, $functions, $globals);
  69.         uasort($elements, array($this, 'compareElements'));
  70.  
  71.         ob_start();
  72.  
  73.         $letter = 64;
  74.         foreach ($elements as $name => $element) {
  75.             $firstChar = strtoupper(substr($element->name(), 0, 1));
  76.             if (is_object($element) && $firstChar != chr($letter)) {
  77.                 $letter = ord($firstChar);
  78.                 echo '<a href="#letter', chr($letter), '">', chr($letter), "</a>\n";
  79.             }
  80.         }
  81.  
  82.         echo "<hr>\n\n";
  83.  
  84.         $first = TRUE;
  85.         foreach ($elements as $element) {
  86.             if (is_object($element)) {
  87.                 if (strtoupper(substr($element->name(), 0, 1)) != chr($letter)) {
  88.                     $letter = ord(strtoupper(substr($element->name(), 0, 1)));
  89.                     if (!$first) {
  90.                         echo "</dl>\n";
  91.                     }
  92.                     $first = FALSE;
  93.                     echo '<h1 id="letter', chr($letter), '">', chr($letter), "</h1>\n";
  94.                     echo "<dl>\n";
  95.                 }
  96.                 $parent =& $element->containingClass();
  97.                 if ($parent && strtolower(get_class($parent)) != 'rootdoc') {
  98.                     $in = 'class <a href="'.$parent->asPath().'">'.$parent->qualifiedName().'</a>';
  99.                 } else {
  100.                     $package =& $element->containingPackage();
  101.                     $in = 'namespace <a href="'.$package->asPath().'/package-summary.html">'.$package->name().'</a>';
  102.                 }
  103.                 switch (strtolower(get_class($element))) {
  104.                 case 'classdoc':
  105.                     if ($element->isOrdinaryClass()) {
  106.                         echo '<dt><a href="', $element->asPath(), '">', $element->name(), '()</a> - Class in ', $in, "</dt>\n";
  107.                     } elseif ($element->isInterface()) {
  108.                         echo '<dt><a href="', $element->asPath(), '">', $element->name(), '()</a> - Interface in ', $in, "</dt>\n";
  109.                     } elseif ($element->isException()) {
  110.                         echo '<dt><a href="', $element->asPath(), '">', $element->name(), '()</a> - Exception in ', $in, "</dt>\n";
  111.                     }
  112.                     break;
  113.                 case 'methoddoc':
  114.                     if ($element->isMethod()) {
  115.                         echo '<dt><a href="', $element->asPath(), '">', $element->name(), '()</a> - Method in ', $in, "</dt>\n";
  116.                     } elseif ($element->isFunction()) {
  117.                         echo '<dt><a href="', $element->asPath(), '">', $element->name(), '()</a> - Function in ', $in, "</dt>\n";
  118.                     }
  119.                     break;
  120.                 case 'fielddoc':
  121.                     if ($element->isGlobal()) {
  122.                         echo '<dt><a href="', $element->asPath(), '">', $element->name(), '()</a> - Global in ', $in, "</dt>\n";
  123.                     }
  124.                     break;
  125.                 }
  126.                 if ($textTag =& $element->tags('@text') && $firstSentenceTags =& $textTag->firstSentenceTags($this->_doclet)) {
  127.                     echo '<dd>';
  128.                     foreach ($firstSentenceTags as $firstSentenceTag) {
  129.                         echo $firstSentenceTag->text($this->_doclet);
  130.                     }
  131.                     echo "</dd>\n";
  132.                 }
  133.             }
  134.         }
  135.         echo "</dl>\n";
  136.  
  137.         $this->_output = ob_get_contents();
  138.         ob_end_clean();
  139.  
  140.         $this->_write('index-all.html', 'Index', TRUE);
  141.  
  142. 	}
  143.  
  144.     function compareElements($element1, $element2)
  145.     {
  146.         $e1 = strtolower($element1->name());
  147.         $e2 = strtolower($element2->name());
  148.         if ($e1 == $e2) {
  149.             return 0;
  150.         } elseif ($e1 < $e2) {
  151.             return -1;
  152.         } else {
  153.             return 1;
  154.         }
  155.     }
  156.  
  157. }
