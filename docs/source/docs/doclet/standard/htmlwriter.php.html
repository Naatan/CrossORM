  66. 	/** Build the HTML header. Includes doctype definition, <html> and <head>
  67. 	 * sections, meta data and window title.
  68. 	 *
  69. 	 * @return str
  70. 	 */
  71. 	function _htmlHeader($title)
  72.     {
  73.  
  74. 		$output = $this->_doctype();
  75. 		$output .= '<html lang="en">'."\n";
  76. 		$output .= "<head>\n\n";
  77.  
  78. 		$output .= '<meta http-equiv="Content-Type" content="text/html; charset=UTF-8">'."\n\n";
  79. 		$output .= '<meta name="generator" content="PHPDoctor '.$this->_doclet->version().' (http://peej.github.com/phpdoctor/)">'."\n";
  80. 		$output .= '<meta name="when" content="'.gmdate('r').'">'."\n\n";
  81.  
  82. 		$output .= '<link rel="stylesheet" type="text/css" href="'.str_repeat('../', $this->_depth).'stylesheet.css">'."\n";
  83. 		$output .= '<link rel="start" href="'.str_repeat('../', $this->_depth).'overview-summary.html">'."\n\n";
  84.  
  85. 		$output .= '<title>';
  86. 		if ($title) {
  87. 			$output .= $title.' ('.$this->_doclet->windowTitle().')';
  88. 		} else {
  89. 			$output .= $this->_doclet->windowTitle();
  90. 		}
  91. 		$output .= "</title>\n\n";
  92. 		$output .= "</head>\n";
  93.  
  94. 		return $output;
  95.  
  96. 	}
  97.  
  98.     /** Get the HTML DOCTYPE for this output
  99.      *
  100.      * @return str
  101.      */
  102.     function _doctype()
  103.     {
  104.         return '<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.01//EN" "http://www.w3.org/TR/html4/strict.dtd">'."\n\n";
  105.     }

  107. 	/** Build the HTML footer.
  108.    *
  109.    * @return str
  110.    */
  111. 	function _htmlFooter()
  112.     {
  113. 		return '</html>';
  114. 	}

  116. 	/** Build the HTML shell header. Includes beginning of the <body> section,
  117. 	 * and the page header.
  118. 	 *
  119. 	 * @return str
  120. 	 */
  121. 	function _shellHeader($path)
  122.     {
  123. 		$output = '<body id="'.$this->_id.'" onload="parent.document.title=document.title;">'."\n\n";
  124. 		$output .= $this->_nav($path);
  125. 		return $output;

  128. 	/** Build the HTML shell footer. Includes the end of the <body> section, and

  130. 	 *
  131. 	 * @return str
  132. 	 */
  133. 	function _shellFooter($path)
  134.     {
  135. 		$output = $this->_nav($path);
  136. 		$output .= "<hr>\n\n";
  137. 		$output .= '<p id="footer">'.$this->_doclet->bottom().'</p>'."\n\n";
  138. 		$output .= "</body>\n\n";
  139. 		return $output;
  140. 	}

  142. 	/** Build the navigation bar
  143. 	 *
  144. 	 * @return str
  145. 	 */
  146. 	function _nav($path)
  147.     {
  148. 		$output = '<div class="header">'."\n";
  149. 		$output .= '<h1>'.$this->_doclet->getHeader()."</h1>\n";
  150. 		if ($this->_sections) {
  151.             $output .= "<ul>\n";
  152. 			foreach ($this->_sections as $section) {
  153. 				if (isset($section['selected']) && $section['selected']) {
  154. 					$output .= '<li class="active">'.$section['title']."</li>\n";
  155. 				} else {
  156. 					if (isset($section['url'])) {
  157. 						$output .= '<li><a href="'.str_repeat('../', $this->_depth).$section['url'].'">'.$section['title']."</a></li>\n";
  158. 					} else {
  159. 						$output .= '<li>'.$section['title'].'</li>';
  160. 					}
  161. 				}
  162. 			}
  163.             $output .= "</ul>\n";
  164. 		}
  165. 		$output .= "</div>\n\n";
  166.  
  167. 		$output .= '<div class="small_links">'."\n";
  168. 		$output .= '<a href="'.str_repeat('../', $this->_depth).'index.html" target="_top">Frames</a>'."\n";
  169. 		$output .= '<a href="'.str_repeat('../', $this->_depth).$path.'" target="_top">No frames</a>'."\n";
  170. 		$output .= "</div>\n";
  171.         $thisClass = strtolower(get_class($this));
  172. 		if ($thisClass == 'classwriter') {
  173. 			$output .= '<div class="small_links">'."\n";
  174. 			$output .= 'Summary: <a href="#summary_field">Field</a> | <a href="#summary_method">Method</a> | <a href="#summary_constr">Constr</a>'."\n";
  175. 			$output .= 'Detail: <a href="#detail_field">Field</a> | <a href="#detail_method">Method</a> | <a href="#summary_constr">Constr</a>'."\n";
  176. 			$output .= "</div>\n";
  177. 		} elseif ($thisClass == 'functionwriter') {
  178. 			$output .= '<div class="small_links">'."\n";
  179. 			$output .= 'Summary: <a href="#summary_function">Function</a>'."\n";
  180. 			$output .= 'Detail: <a href="#detail_function">Function</a>'."\n";
  181. 			$output .= "</div>\n";
  182. 		} elseif ($thisClass == 'globalwriter') {
  183. 			$output .= '<div class="small_links">'."\n";
  184. 			$output .= 'Summary: <a href="#summary_global">Global</a>'."\n";
  185. 			$output .= 'Detail: <a href="#detail_global">Global</a>'."\n";
  186. 			$output .= "</div>\n";
  187. 		}
  188.  
  189. 		return $output;
  190. 	}

  192. 	function _sourceLocation($doc)
  193. 	{
  194. 	    if ($this->_doclet->includeSource()) {
  195. 	        $url = strtolower(str_replace(DIRECTORY_SEPARATOR, '/', $doc->sourceFilename()));
  196. 	        echo '<a href="', str_repeat('../', $this->_depth), 'source/', $url, '.html#line', $doc->sourceLine(), '" class="location">', $doc->location(), "</a>\n\n";
  197. 	    } else {
  198. 	        echo '<div class="location">', $doc->location(), "</div>\n";
  199. 	    }
  200. 	}

  202. 	/** Write the HTML page to disk using the given path.
  203. 	 *
  204. 	 * @param str path The path to write the file to
  205. 	 * @param str title The title for this page
  206. 	 * @param bool shell Include the page shell in the output
  207. 	 */
  208. 	function _write($path, $title, $shell)
  209.     {
  210. 		$phpdoctor =& $this->_doclet->phpdoctor();
  211.  
  212. 		// make directory separators suitable to this platform
  213. 		$path = str_replace('/', DIRECTORY_SEPARATOR, $path);
  214.  
  215. 		// make directories if they don't exist
  216. 		$dirs = explode(DIRECTORY_SEPARATOR, $path);
  217. 		array_pop($dirs);
  218. 		$testPath = $this->_doclet->destinationPath();
  219. 		foreach ($dirs as $dir) {
  220. 			$testPath .= $dir.DIRECTORY_SEPARATOR;
  221. 			if (!is_dir($testPath)) {
  222.                 if (!@mkdir($testPath)) {
  223.                     $phpdoctor->error(sprintf('Could not create directory "%s"', $testPath));
  224.                     exit;
  225.                 }
  226.             }
  227. 		}
  228.  
  229. 		// write file
  230. 		$fp = fopen($this->_doclet->destinationPath().$path, 'w');
  231. 		if ($fp) {
  232. 			$phpdoctor->message('Writing "'.$path.'"');
  233. 			fwrite($fp, $this->_htmlHeader($title));
  234. 			if ($shell) fwrite($fp, $this->_shellHeader($path));
  235. 			fwrite($fp, $this->_output);
  236. 			if ($shell) fwrite($fp, $this->_shellFooter($path));
  237. 			fwrite($fp, $this->_htmlFooter());
  238. 			fclose($fp);
  239. 		} else {
  240. 			$phpdoctor->error('Could not write "'.$this->_doclet->destinationPath().$path.'"');
  241.             exit;
  242. 		}
  243. 	}

  245. 	/** Format tags for output.
  246. 	 *
  247. 	 * @param Tag[] tags
  248. 	 * @return str The string representation of the elements doc tags
  249. 	 */
  250. 	function _processTags(&$tags)
  251.     {
  252. 		$tagString = '';
  253. 		foreach ($tags as $key => $tag) {
  254. 			if ($key != '@text') {
  255. 			    if (is_array($tag)) {
  256. 				    $hasText = FALSE;
  257.                     foreach ($tag as $key => $tagFromGroup) {
  258.                         if ($tagFromGroup->text($this->_doclet) != '') {
  259.                             $hasText = TRUE;
  260.                         }
  261.                     }
  262.                     if ($hasText) {
  263.                         $tagString .= '<dt>'.$tag[0]->displayName().":</dt>\n";
  264.                         foreach ($tag as $tagFromGroup) {
  265.                             $tagString .= '<dd>'.$tagFromGroup->text($this->_doclet)."</dd>\n";
  266.                         }
  267.                     }
  268. 				} else {
  269. 			        $text = $tag->text($this->_doclet);
  270. 			        if ($text != '') {
  271. 						$tagString .= '<dt>'.$tag->displayName().":</dt>\n";
  272. 						$tagString .= '<dd>'.$text."</dd>\n";
  273. 					} elseif ($tag->displayEmpty()) {
  274. 						$tagString .= '<dt>'.$tag->displayName().".</dt>\n";
  275. 					}
  276. 				}
  277. 			}
  278. 		}
  279.         if ($tagString) {
  280.             echo "<dl>\n", $tagString, "</dl>\n";
  281.         }
  282. 	}

  284. 	/** Convert inline tags into a string for outputting.
  285. 	 *
  286. 	 * @param Tag tag The text tag to process
  287. 	 * @param bool first Process first line of tag only
  288. 	 * @return str The string representation of the elements doc tags
  289. 	 */
  290. 	function _processInlineTags(&$tag, $first = FALSE)
  291.     {
  292.         $description = '';
  293.         if (is_array($tag)) $tag = $tag[0];
  294.         if (is_object($tag)) {
  295.             if ($first) {
  296.                 $tags =& $tag->firstSentenceTags($this->_doclet);
  297.             } else {
  298.                 $tags =& $tag->inlineTags($this->_doclet);
  299.             }
  300.             if ($tags) {
  301.                 foreach ($tags as $aTag) {
  302.                     if ($aTag) {
  303.                         $description .= $aTag->text($this->_doclet);
  304.                     }
  305.                 }
  306.             }
  307.             return $this->_doclet->formatter->toFormattedText($description);
  308. 		}
  309.         return NULL;
  310. 	}
