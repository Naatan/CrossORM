  21. /** This generates the package-frame.html file that lists the interfaces and
  22.  * classes in a given package for displaying in the lower-left frame of the
  23.  * frame-formatted default output.
  24.  *

  26.  */
  27. class PackageFrameWriter extends HTMLWriter

  30. 	/** Build the package frame index.
  31. 	 *
  32. 	 * @param Doclet doclet
  33. 	 */
  34. 	function packageFrameWriter(&$doclet)
  35.     {
  36.  
  37. 		parent::HTMLWriter($doclet);
  38.  
  39. 		$rootDoc =& $this->_doclet->rootDoc();
  40.  
  41. 		$this->_output =& $this->_allItems($rootDoc);
  42. 		$this->_write('allitems-frame.html', 'All Items', FALSE);
  43.  
  44.         $packages =& $rootDoc->packages();
  45.         ksort($packages);
  46.  
  47. 		foreach($packages as $packageName => $package) {
  48.  
  49. 			$this->_depth = $package->depth() + 1;
  50.  
  51. 			$this->_output =& $this->_buildFrame($package);
  52. 			$this->_write($package->asPath().'/package-frame.html', $package->name(), FALSE);
  53.  
  54. 		}
  55.  
  56. 	}
  57.  
  58. 	/** Build package frame
  59. 	 *
  60. 	 * @return str
  61. 	 */
  62. 	function &_buildFrame(&$package)

  65.  
  66. 		echo '<body id="frame">', "\n\n";
  67.  
  68. 		echo '<h1><a href="package-summary.html" target="main">', $package->name(), "</a></h1>\n\n";
  69.  
  70. 		$classes =& $package->ordinaryClasses();
  71. 		if ($classes && is_array($classes)) {
  72.             ksort($classes);
  73. 			echo "<h2>Classes</h2>\n";
  74. 			echo "<ul>\n";
  75. 			foreach($classes as $name => $class) {
  76. 				echo '<li><a href="', str_repeat('../', $package->depth() + 1), $classes[$name]->asPath(), '" target="main">', $classes[$name]->name(), "</a></li>\n";
  77. 			}
  78. 			echo "</ul>\n\n";
  79. 		}
  80.  
  81. 		$interfaces =& $package->interfaces();

  83.             ksort($interfaces);
  84. 			echo "<h2>Interfaces</h2>\n";
  85. 			echo "<ul>\n";
  86. 			foreach($interfaces as $name => $interface) {
  87. 				echo '<li><a href="', str_repeat('../', $package->depth() + 1), $interfaces[$name]->asPath(), '" target="main">', $interfaces[$name]->name(), "</a></li>\n";
  88. 			}
  89. 			echo "</ul>\n\n";
  90. 		}
  91.  
  92. 		$exceptions =& $package->exceptions();
  93. 		if ($exceptions && is_array($exceptions)) {
  94.             ksort($exceptions);
  95. 			echo "<h2>Exceptions</h2>\n";
  96. 			echo "<ul>\n";
  97. 			foreach($exceptions as $name => $exception) {
  98. 				echo '<li><a href="', str_repeat('../', $package->depth() + 1), $exceptions[$name]->asPath(), '" target="main">', $exceptions[$name]->name(), "</a></li>\n";
  99. 			}
  100. 			echo "</ul>\n\n";
  101. 		}
  102.  
  103. 		$functions =& $package->functions();

  110. 			}
  111. 			echo "</ul>\n\n";
  112. 		}
  113.

  120. 				echo '<li><a href="', str_repeat('../', $package->depth() + 1), $globals[$name]->asPath(), '" target="main">', $globals[$name]->name(), "</a></li>\n";
  121. 			}
  122. 			echo "</ul>\n\n";
  123. 		}
  124.  
  125. 		echo '</body>', "\n\n";
  126.  
  127. 		$output = ob_get_contents();
  128. 		ob_end_clean();

  130. 		return $output;
  131. 	}

  133. 	/** Build all items frame
  134. 	 *
  135. 	 * @return str
  136. 	 */
  137. 	function &_allItems(&$rootDoc)
  138.     {
  139. 		ob_start();
  140.  
  141. 		echo '<body id="frame">', "\n\n";
  142.  
  143. 		echo "<h1>All Items</h1>\n\n";
  144.  
  145. 		$classes =& $rootDoc->classes();
  146. 		if ($classes) {
  147.             ksort($classes);
  148. 			echo "<h2>Classes</h2>\n";
  149. 			echo "<ul>\n";
  150. 			foreach($classes as $name => $class) {
  151. 				$package =& $classes[$name]->containingPackage();
  152. 				if ($class->isInterface()) {
  153. 					echo '<li><em><a href="', $classes[$name]->asPath(), '" title="', $classes[$name]->packageName(),'" target="main">', $classes[$name]->name(), "</a></em></li>\n";
  154. 				} else {
  155. 					echo '<li><a href="', $classes[$name]->asPath(), '" title="', $classes[$name]->packageName(),'" target="main">', $classes[$name]->name(), "</a></li>\n";
  156. 				}
  157. 			}
  158. 			echo "</ul>\n\n";
  159. 		}
  160.  
  161. 		$functions =& $rootDoc->functions();
  162. 		if ($functions) {
  163.             ksort($functions);
  164. 			echo "<h2>Functions</h2>\n";
  165. 			echo "<ul>\n";
  166. 			foreach($functions as $name => $function) {
  167. 				$package =& $functions[$name]->containingPackage();
  168. 				echo '<li><a href="', $package->asPath(), '/package-functions.html#', $functions[$name]->name(), '()" title="', $functions[$name]->packageName(),'" target="main">', $functions[$name]->name(), "</a></li>\n";
  169. 			}
  170. 			echo "</ul>\n\n";
  171. 		}
  172.  
  173. 		$globals =& $rootDoc->globals();

  179. 				$package =& $globals[$name]->containingPackage();
  180. 				echo '<li><a href="', $package->asPath(), '/package-globals.html#', $globals[$name]->name(), '" title="', $globals[$name]->packageName(),'" target="main">', $globals[$name]->name(), "</a></li>\n";
  181. 			}
