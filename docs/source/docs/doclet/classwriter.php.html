  327. 	/** Build the class hierarchy tree which is placed at the top of the page.
  328. 	 *
  329. 	 * @param RootDoc rootDoc The root doc
  330. 	 * @param ClassDoc class Class to generate tree for
  331. 	 * @param int depth Depth of recursion
  332. 	 * @return mixed[]
  333. 	 */
  334. 	function _buildTree(&$rootDoc, &$class, $depth = NULL)
  335.     {
  336. 		if ($depth === NULL) {
  337. 			$start = TRUE;
  338. 			$depth = 0;
  339. 		} else {
  340. 			$start = FALSE;
  341. 		}
  342. 		$output = '';
  343. 		$undefinedClass = FALSE;
  344. 		if ($class->superclass()) {
  345. 			$superclass =& $rootDoc->classNamed($class->superclass());
  346. 			if ($superclass) {
  347. 				$result = $this->_buildTree($rootDoc, $superclass, $depth);
  348. 				$output .= $result[0];
  349. 				$depth = ++$result[1];
  350. 			} else {
  351. 				$output .= $class->superclass().'<br>';
  352. 				$output .= str_repeat('   ', $depth).' ';
  353. 				$depth++;
  354. 				$undefinedClass = TRUE;
  355. 			}
  356. 		}
  357. 		if ($depth > 0 && !$undefinedClass) {
  358. 			$output .= str_repeat('   ', $depth).' ';
  359. 		}
  360. 		if ($start) {
  361. 			$output .= '<strong>'.$class->name().'</strong><br />';
  362. 		} else {
  363. 			$output .= '<a href="'.str_repeat('../', $this->_depth).$class->asPath().'">'.$class->name().'</a><br>';
  364. 		}
  365. 		return array($output, $depth);
  366. 	}
