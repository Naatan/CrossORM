  109. 	function standard(&$rootDoc, $formatter)
  110.     {
  111.  
  112. 		// set doclet options
  113. 		$this->_rootDoc =& $rootDoc;
  114. 		$phpdoctor =& $rootDoc->phpdoctor();
  115. 		$options =& $rootDoc->options();
  116.  
  117. 		$this->formatter = $formatter;
  118.  
  119. 		if (isset($options['d'])) {
  120. 			$this->_d = $phpdoctor->makeAbsolutePath($options['d'], $phpdoctor->sourcePath());
  121. 		} elseif (isset($options['output_dir'])) {
  122. 			$this->_d = $phpdoctor->makeAbsolutePath($options['output_dir'], $phpdoctor->sourcePath());
  123. 		} else {
  124. 			$this->_d = $phpdoctor->makeAbsolutePath('apidocs', $phpdoctor->sourcePath());
  125. 		}
  126. 		$this->_d = $phpdoctor->fixPath($this->_d);
  127.  
  128. 		if (is_dir($this->_d)) {
  129. 			$phpdoctor->warning('Output directory already exists, overwriting');
  130. 		} else {
  131. 			mkdir($this->_d);
  132. 		}
  133. 		$phpdoctor->verbose('Setting output directory to "'.$this->_d.'"');
  134.  
  135. 		if (isset($options['windowtitle'])) $this->_windowTitle = $options['windowtitle'];
  136. 		if (isset($options['doctitle'])) $this->_docTitle = $options['doctitle'];
  137. 		if (isset($options['header'])) $this->_header = $options['header'];
  138. 		if (isset($options['footer'])) $this->_footer = $options['footer'];
  139. 		if (isset($options['bottom'])) $this->_bottom = $options['bottom'];
  140.  
  141. 		if (isset($options['tree'])) $this->_tree = $options['tree'];
  142.  
  143. 		if (isset($options['include_source'])) $this->_includeSource = $options['include_source'];
  144.         if ($this->_includeSource) {
  145.             @include_once 'geshi/geshi.php';
  146.             if (!class_exists('GeSHi')) {
  147.                 $phpdoctor->warning('Could not find GeSHi in "geshi/geshi.php", not pretty printing source');
  148.             }
  149. 		}
  150.  
  151. 		// write frame
  152. 		$frameOutputWriter =& new frameOutputWriter($this);
  153.  
  154. 		// write overview summary
  155. 		$packageIndexWriter =& new packageIndexWriter($this);
  156.  
  157. 		// write package overview frame
  158. 		$packageIndexFrameWriter =& new packageIndexFrameWriter($this);
  159.  
  160. 		// write package summaries
  161. 		$packageWriter =& new packageWriter($this);
  162.  
  163. 		// write package frame
  164. 		$packageFrameWriter =& new packageFrameWriter($this);
  165.  
  166. 		// write classes
  167. 		$classWriter =& new classWriter($this);
  168.  
  169. 		// write global functions
  170. 		$functionWriter =& new functionWriter($this);
  171.  
  172. 		// write global variables
  173. 		$globalWriter =& new globalWriter($this);
  174.  
  175. 		// write index
  176. 		$indexWriter =& new indexWriter($this);
  177.  
  178. 		// write deprecated index
  179. 		$deprecatedWriter =& new deprecatedWriter($this);
  180.  
  181. 		// write todo index

  183.

  185. 		if ($this->_includeSource) {
  186.             $sourceWriter =& new sourceWriter($this);
  187. 		}

  189. 		// copy stylesheet
  190. 		$phpdoctor->message('Copying stylesheet');
  191. 		copy($phpdoctor->docletPath().'stylesheet.css', $this->_d.'stylesheet.css');

  193. 	}
