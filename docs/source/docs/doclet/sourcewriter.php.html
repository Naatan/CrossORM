  36. 		parent::HTMLWriter($doclet);
  37.  
  38. 		$rootDoc =& $this->_doclet->rootDoc();
  39. 		$phpdoctor =& $this->_doclet->phpdoctor();
  40.  
  41.         $this->_sections[0] = array('title' => 'Overview', 'url' => 'overview-summary.html');

  67.

  73.  
  74. 		$this->_id = 'file';
  75.

  77.  
  78. 			$this->_sections[0] = array('title' => 'Overview', 'url' => 'overview-summary.html');
  79. 			$this->_sections[1] = array('title' => 'Package');
  80. 			$this->_sections[2] = array('title' => 'Class');
  81. 			//$this->_sections[3] = array('title' => 'Use');
  82. 			if ($phpdoctor->getOption('tree')) $this->_sections[4] = array('title' => 'Tree');
  83. 			$this->_sections[5] = array('title' => 'Files', 'url' => 'overview-files.html');

  88.             $this->_depth = substr_count($filename, '/') + 1;

  97.  
  98.             ob_start();
  99.

  105.             }
  106.

  108.  
  109.             foreach (explode("\n", $source) as $index => $line) {

  111.             }
  112.  
  113.             $this->_output = ob_get_contents();
  114.             ob_end_clean();
  115.  
  116.             $this->_write('source/'.strtolower($filename).'.html', $filename, TRUE);
  117.  
  118. 		}

  120.     }
