  237. 	/**
  238. 	 * Build the class tree branch for the given element
  239. 	 *
  240. 	 * @param ClassDoc[] tree
  241. 	 * @param ClassDoc element
  242. 	 */
  243. 	function _buildTree(&$tree, &$element)
  244.     {
  245. 		$tree[$element->name()] = $element;
  246. 		if ($element->superclass()) {
  247. 			$rootDoc =& $this->_doclet->rootDoc();
  248.             $superclass =& $rootDoc->classNamed($element->superclass());
  249.             if ($superclass) {
  250.                 $this->_buildTree($tree, $superclass);
  251.             }
  252. 		}
  253. 	}
  254.  
  255. 	/**
  256. 	 * Build the class tree branch for the given element
  257. 	 *
  258. 	 * @param ClassDoc[] tree
  259. 	 * @param str parent
  260. 	 */
  261. 	function _displayTree($tree, $parent = NULL)
  262.     {
  263. 		$outputList = TRUE;
  264. 		foreach($tree as $name => $element) {
  265. 			if ($element->superclass() == $parent) {
  266. 				if ($outputList) echo "<ul>\n";
  267. 				echo '<li><a href="', str_repeat('../', $this->_depth), $element->asPath(), '">', $element->qualifiedName(), '</a>';
  268. 				$this->_displayTree($tree, $name);
  269. 				echo "</li>\n";
  270. 				$outputList = FALSE;
  271. 			}
  272. 		}
  273. 		if (!$outputList) echo "</ul>\n";
  274. 	}
