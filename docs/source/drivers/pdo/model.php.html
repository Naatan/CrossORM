  93. 	* Static method to get a table name given a class name.
  94. 	* If the supplied class has a public static property

  146. 	/**
  147. 	* Factory method used to acquire instances of the given class.
  148. 	* The class name should be supplied as a string, and the class
  149. 	* should already have been loaded by PHP (or a suitable autoloader
  150. 	* should exist). This method actually returns a wrapped ORM object
  151. 	* which allows a database query to be built. The wrapped ORM object is
  152. 	* responsible for returning instances of the correct class when
  153. 	* its find_one or find_many methods are called.
  154. 	*/
  155. 	public static function factory($class_name)
  156. 	{
  157. 		$table_name = self::_get_table_name($class_name);
  158. 		$wrapper 	= ORMWrapper::for_table($table_name);
  159.  
  160. 		$wrapper->set_class_name($class_name);
  161. 		$wrapper->use_id_column(self::_get_id_column_name($class_name));
  162.  
  163. 		return $wrapper;
  164. 	}

  167. 	* Internal method to construct the queries for both the has_one and
  168. 	* has_many methods. These two types of association are identical; the
  169. 	* only difference is whether find_one or find_many is used to complete
  170. 	* the method chain.
  171. 	*/
  172. 	protected function _has_one_or_many($associated_class_name, $foreign_key_name=null)
  173. 	{
  174. 		$base_table_name 	= self::_get_table_name(get_class($this));
  175. 		$foreign_key_name 	= self::_build_foreign_key_name($foreign_key_name, $base_table_name);
  176.  
  177. 		return self::factory($associated_class_name)->where($foreign_key_name, $this->id());

  212. 	* Helper method to manage many-to-many relationships via an intermediate model. See
  213. 	* README for a full explanation of the parameters.
  214. 	*/
  215. 	public function has_many_through($associated_class_name, $join_class_name=null, $key_to_base_table=null, $key_to_associated_table=null)
  216. 	{
  217. 		$base_class_name = get_class($this);
  218.  
  219. 		// The class name of the join model, if not supplied, is
  220. 		// formed by concatenating the names of the base class
  221. 		// and the associated class, in alphabetical order.
  222. 		if (is_null($join_class_name)) {
  223. 			$class_names 			= array($base_class_name, $associated_class_name);
  224. 			sort($class_names, SORT_STRING);
  225. 			$join_class_name 		= join("", $class_names);
  226. 		}
  227.  
  228. 		// Get table names for each class
  229. 		$base_table_name 			= self::_get_table_name($base_class_name);
  230. 		$associated_table_name 		= self::_get_table_name($associated_class_name);
  231. 		$join_table_name 			= self::_get_table_name($join_class_name);
  232.  
  233. 		// Get ID column names
  234. 		$base_table_id_column 		= self::_get_id_column_name($base_class_name);
  235. 		$associated_table_id_column = self::_get_id_column_name($associated_class_name);
  236.  
  237. 		// Get the column names for each side of the join table
  238. 		$key_to_base_table 			= self::_build_foreign_key_name($key_to_base_table, $base_table_name);
  239. 		$key_to_associated_table 	= self::_build_foreign_key_name($key_to_associated_table, $associated_table_name);
  240.  
  241. 		return self::factory($associated_class_name)
  242. 			->select("{$associated_table_name}.*")
  243. 			->join($join_table_name, array("{$associated_table_name}.{$associated_table_id_column}", '=', "{$join_table_name}.{$key_to_associated_table}"))
  244. 			->where("{$join_table_name}.{$key_to_base_table}", $this->id());
  245. 	}
